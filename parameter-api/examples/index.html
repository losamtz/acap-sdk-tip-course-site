main() - Application Entry Point

1int main(void) {
2    GError* error   = NULL;
3    GMainLoop* loop = NULL;
4
5    openlog(APP_NAME, LOG_PID, LOG_USER);
6    syslog(LOG_INFO, "Starting %s", APP_NAME);
7
8    // 1. Create a new AXParameter handle.
9    AXParameter* handle = ax_parameter_new(APP_NAME, &error);
10    if(handle == NULL)
11        panic("%s", error->message);
12
13    syslog(LOG_INFO, "Starting handle");
14
15    // 2. Act on changes to ParameterManifest as soon as they happen.
16    if(!ax_parameter_register_callback(handle, "ParameterManifest", acap_parameter_changed, NULL, &error))
17        panic("%s", error->message);
18    
19    // 3. Start listening to callbacks by launching a GLib main loop.
20    loop = g_main_loop_new(NULL, FALSE);
21
22    g_unix_signal_add(SIGTERM, signal_handler, loop);
23    g_unix_signal_add(SIGINT, signal_handler, loop);
24    g_main_loop_run(loop);
25
26    g_main_loop_unref(loop);
27    ax_parameter_free(handle);
28}

Key Features:

  • Simple parameter callback registration
  • Automatic parameter creation from manifest
  • Built-in parameter validation
  • Clean signal handling and resource cleanup
