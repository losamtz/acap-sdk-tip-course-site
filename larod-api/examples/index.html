Basic Model Loading and Inference

Simple example showing how to load a model and run inference with static data.

#include <larod.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main() {
    larodError* error = NULL;
    larodConnection* conn = NULL;
    larodModel* model = NULL;
    
    // Connect to Larod service
    conn = larodConnect(&error);
    if (!conn) {
        fprintf(stderr, "Failed to connect to Larod: %s\n", 
                larodGetErrorString(error));
        return 1;
    }
    
    printf("Connected to Larod successfully\n");
    
    // Load TensorFlow Lite model
    const char* model_path = "/usr/local/packages/MyApp/model.tflite";
    model = larodLoadModel(conn, model_path, 
                          LAROD_DEVICE_ID_CPU, 
                          LAROD_ACCESS_PRIVATE, 
                          "", NULL, &error);
    
    if (!model) {
        fprintf(stderr, "Failed to load model: %s\n", 
                larodGetErrorString(error));
        larodDisconnect(&conn, NULL);
        return 1;
    }
    
    printf("Model loaded successfully\n");
    
    // Get model information
    size_t num_inputs = larodGetModelNumInputs(model, &error);
    size_t num_outputs = larodGetModelNumOutputs(model, &error);
    
    printf("Model has %zu inputs and %zu outputs\n", num_inputs, num_outputs);
    
    // Create input tensor (assuming single input)
    larodTensor* input = larodCreateModelInputTensor(model, 0, &error);
    if (!input) {
        fprintf(stderr, "Failed to create input tensor\n");
        goto cleanup;
    }
    
    // Get tensor information
    size_t input_size = larodGetTensorSize(input, &error);
    printf("Input tensor size: %zu bytes\n", input_size);
    
    // Fill input tensor with sample data
    void* input_data = larodGetTensorBuffer(input, &error);
    if (input_data) {
        // Fill with dummy data (normally you'd load actual image data)
        memset(input_data, 128, input_size); // Fill with gray values
        printf("Input tensor filled with sample data\n");
    }
    
    // Create output tensor
    larodTensor* output = larodCreateModelOutputTensor(model, 0, &error);
    if (!output) {
        fprintf(stderr, "Failed to create output tensor\n");
        goto cleanup;
    }
    
    // Prepare tensor arrays for inference
    larodTensor* inputs[] = {input};
    larodTensor* outputs[] = {output};
    
    // Run inference
    printf("Running inference...\n");
    if (!larodRunInference(model, inputs, 1, outputs, 1, &error)) {
        fprintf(stderr, "Inference failed: %s\n", 
                larodGetErrorString(error));
        goto cleanup;
    }
    
    printf("Inference completed successfully!\n");
    
    // Get output data
    void* output_data = larodGetTensorBuffer(output, &error);
    size_t output_size = larodGetTensorSize(output, &error);
    
    printf("Output tensor size: %zu bytes\n", output_size);
    
    // Process output (example: print first few values as floats)
    if (output_data && output_size >= sizeof(float)) {
        float* float_output = (float*)output_data;
        size_t num_floats = output_size / sizeof(float);
        
        printf("First few output values:\n");
        for (size_t i = 0; i < (num_floats < 10 ? num_floats : 10); i++) {
            printf("  [%zu]: %.6f\n", i, float_output[i]);
        }
    }

cleanup:
    // Clean up resources
    if (input) larodDestroyTensor(&input, NULL);
    if (output) larodDestroyTensor(&output, NULL);
    if (model) larodUnloadModel(&model, NULL);
    if (conn) larodDisconnect(&conn, NULL);
    
    printf("Cleanup completed\n");
    return 0;
}
