Lab 1
Hello Larod
Basic
Learn the fundamentals of Larod API by loading a model and running basic inference
Objective: Load a TensorFlow Lite model and perform basic inference with static input data

Key Concepts:

  • Larod connection and device enumeration
  • Model loading and validation
  • Tensor creation and data handling
  • Basic inference execution

Implementation:

#include <larod.h>
#include <stdio.h>
#include <stdlib.h>

int main() {
    larodError* error = NULL;
    larodConnection* conn = NULL;
    larodModel* model = NULL;
    
    // Connect to Larod
    conn = larodConnect(&error);
    if (!conn) {
        fprintf(stderr, "Failed to connect to Larod: %s\n", 
                larodGetErrorString(error));
        return 1;
    }
    
    // Load model
    model = larodLoadModel(conn, "/path/to/model.tflite", 
                          LAROD_DEVICE_ID_CPU, LAROD_ACCESS_PRIVATE, 
                          "", NULL, &error);
    if (!model) {
        fprintf(stderr, "Failed to load model: %s\n", 
                larodGetErrorString(error));
        larodDisconnect(&conn, NULL);
        return 1;
    }
    
    // Create input tensor
    larodTensor* input = larodCreateModelInputTensor(model, 0, &error);
    if (!input) {
        fprintf(stderr, "Failed to create input tensor\n");
        goto cleanup;
    }
    
    // Fill input with data (example: 224x224x3 RGB image)
    void* input_data = larodGetTensorBuffer(input, &error);
    // ... fill input_data with your image data ...
    
    // Create output tensor
    larodTensor* output = larodCreateModelOutputTensor(model, 0, &error);
    
    // Run inference
    larodTensor* inputs[] = {input};
    larodTensor* outputs[] = {output};
    
    if (!larodRunInference(model, inputs, 1, outputs, 1, &error)) {
        fprintf(stderr, "Inference failed: %s\n", 
                larodGetErrorString(error));
        goto cleanup;
    }
    
    // Process output
    void* output_data = larodGetTensorBuffer(output, &error);
    size_t output_size = larodGetTensorSize(output, &error);
    
    printf("Inference completed successfully!\n");
    printf("Output size: %zu bytes\n", output_size);
    
cleanup:
    larodDestroyTensor(&input, NULL);
    larodDestroyTensor(&output, NULL);
    larodUnloadModel(&model, NULL);
    larodDisconnect(&conn, NULL);
    
    return 0;
}
Lab 2
Preprocessing Pipeline
Intermediate
Use Larod's built-in preprocessing for efficient format conversion and normalization
Objective: Implement preprocessing pipeline for RGB to tensor conversion with normalization

Preprocessing Features:

  • Format conversion (RGB, YUV, NV12)
  • Resize and crop operations
  • Normalization and scaling
  • Hardware-accelerated processing

Implementation:

#include <larod.h>

// Create preprocessing configuration
larodPreprocessingConfig* create_preprocess_config(larodConnection* conn) {
    larodError* error = NULL;
    
    larodPreprocessingConfig* config = 
        larodCreatePreprocessingConfig(LAROD_COLORSPACE_RGB, &error);
    
    // Set input dimensions (original image)
    larodSetPreprocessingConfigInputDims(config, 640, 480, &error);
    
    // Set output dimensions (model input)
    larodSetPreprocessingConfigOutputDims(config, 224, 224, &error);
    
    // Set normalization parameters
    float mean[] = {0.485f, 0.456f, 0.406f};  // ImageNet means
    float std[] = {0.229f, 0.224f, 0.225f};   // ImageNet stds
    
    larodSetPreprocessingConfigNormalization(config, mean, std, &error);
    
    // Set crop parameters (center crop)
    larodSetPreprocessingConfigCropParams(config, 
        LAROD_CROP_CENTER, 0, 0, 0, 0, &error);
    
    return config;
}

int run_inference_with_preprocessing(larodConnection* conn, 
                                   larodModel* model,
                                   uint8_t* rgb_data,
                                   size_t rgb_size) {
    larodError* error = NULL;
    
    // Create preprocessing config
    larodPreprocessingConfig* preprocess_config = 
        create_preprocess_config(conn);
    
    // Create input tensor from raw RGB data
    larodTensor* raw_input = larodCreateTensor(conn, rgb_size, 
                                              LAROD_TENSOR_LAYOUT_NHWC,
                                              LAROD_TENSOR_DATA_TYPE_UINT8,
                                              &error);
    
    // Copy RGB data to tensor
    void* tensor_data = larodGetTensorBuffer(raw_input, &error);
    memcpy(tensor_data, rgb_data, rgb_size);
    
    // Create preprocessed tensor
    larodTensor* preprocessed = larodCreateModelInputTensor(model, 0, &error);
    
    // Apply preprocessing
    if (!larodRunPreprocessing(conn, preprocess_config, 
                              raw_input, preprocessed, &error)) {
        fprintf(stderr, "Preprocessing failed: %s\n", 
                larodGetErrorString(error));
        return -1;
    }
    
    // Create output tensor
    larodTensor* output = larodCreateModelOutputTensor(model, 0, &error);
    
    // Run inference with preprocessed input
    larodTensor* inputs[] = {preprocessed};
    larodTensor* outputs[] = {output};
    
    if (!larodRunInference(model, inputs, 1, outputs, 1, &error)) {
        fprintf(stderr, "Inference failed\n");
        return -1;
    }
    
    printf("Inference with preprocessing completed!\n");
    
    // Cleanup
    larodDestroyTensor(&raw_input, NULL);
    larodDestroyTensor(&preprocessed, NULL);
    larodDestroyTensor(&output, NULL);
    larodDestroyPreprocessingConfig(&preprocess_config, NULL);
    
    return 0;
}
Lab 3
VDO-Larod Integration
Advanced
Real-time inference on video frames using VDO and Larod integration
Objective: Process live video frames with machine learning inference in real-time

Integration Points:

  • VDO stream configuration for ML input
  • Frame buffer to tensor conversion
  • Asynchronous inference processing
  • Result synchronization with video pipeline

Implementation:

#include <vdo-stream.h>
#include <larod.h>
#include <glib.h>

typedef struct {
    VdoStream* stream;
    larodConnection* larod_conn;
    larodModel* model;
    larodPreprocessingConfig* preprocess_config;
    GMainLoop* loop;
} AppContext;

// Frame callback for VDO stream
static void frame_callback(VdoBuffer* buffer, void* user_data) {
    AppContext* ctx = (AppContext*)user_data;
    larodError* error = NULL;
    
    // Get frame data
    VdoMap* map = vdo_buffer_get_data(buffer);
    uint8_t* frame_data = vdo_map_get_data(map);
    size_t frame_size = vdo_map_get_size(map);
    
    // Create tensor from frame data
    larodTensor* input_tensor = larodCreateTensor(ctx->larod_conn,
                                                 frame_size,
                                                 LAROD_TENSOR_LAYOUT_NHWC,
                                                 LAROD_TENSOR_DATA_TYPE_UINT8,
                                                 &error);
    
    // Copy frame data to tensor
    void* tensor_data = larodGetTensorBuffer(input_tensor, &error);
    memcpy(tensor_data, frame_data, frame_size);
    
    // Create preprocessed tensor
    larodTensor* preprocessed = larodCreateModelInputTensor(ctx->model, 0, &error);
    
    // Apply preprocessing
    if (larodRunPreprocessing(ctx->larod_conn, ctx->preprocess_config,
                             input_tensor, preprocessed, &error)) {
        
        // Create output tensor
        larodTensor* output = larodCreateModelOutputTensor(ctx->model, 0, &error);
        
        // Run inference
        larodTensor* inputs[] = {preprocessed};
        larodTensor* outputs[] = {output};
        
        if (larodRunInference(ctx->model, inputs, 1, outputs, 1, &error)) {
            // Process inference results
            process_inference_results(output);
        }
        
        larodDestroyTensor(&output, NULL);
    }
    
    // Cleanup
    larodDestroyTensor(&input_tensor, NULL);
    larodDestroyTensor(&preprocessed, NULL);
    vdo_map_unref(map);
}

int main() {
    AppContext ctx = {0};
    GError* error = NULL;
    
    // Initialize VDO stream
    ctx.stream = vdo_stream_new(&error);
    vdo_stream_set_format(ctx.stream, VDO_FORMAT_YUV);
    vdo_stream_set_resolution(ctx.stream, 640, 480);
    vdo_stream_set_framerate(ctx.stream, 30);
    
    // Set frame callback
    vdo_stream_set_buffer_callback(ctx.stream, frame_callback, &ctx);
    
    // Initialize Larod
    larodError* larod_error = NULL;
    ctx.larod_conn = larodConnect(&larod_error);
    ctx.model = larodLoadModel(ctx.larod_conn, "/path/to/model.tflite",
                              LAROD_DEVICE_ID_CPU, LAROD_ACCESS_PRIVATE,
                              "", NULL, &larod_error);
    
    // Create preprocessing config
    ctx.preprocess_config = create_preprocess_config(ctx.larod_conn);
    
    // Start stream
    vdo_stream_start(ctx.stream, &error);
    
    // Run main loop
    ctx.loop = g_main_loop_new(NULL, FALSE);
    g_main_loop_run(ctx.loop);
    
    // Cleanup
    vdo_stream_stop(ctx.stream);
    g_object_unref(ctx.stream);
    larodUnloadModel(&ctx.model, NULL);
    larodDisconnect(&ctx.larod_conn, NULL);
    g_main_loop_unref(ctx.loop);
    
    return 0;
}
Lab 4
Output Processing & Overlay
Advanced
Parse inference results and visualize them using overlay API
Objective: Process model outputs and create visual overlays for object detection results

Output Processing:

  • Parse detection results (bounding boxes, classes, scores)
  • Apply confidence thresholds and NMS
  • Convert coordinates to overlay space
  • Draw bounding boxes and labels

Implementation:

#include <larod.h>
#include <axoverlay.h>

typedef struct {
    float x, y, width, height;
    int class_id;
    float confidence;
    char class_name[64];
} Detection;

// Parse YOLO-style output tensor
int parse_detections(larodTensor* output, Detection* detections, 
                    int max_detections, float confidence_threshold) {
    larodError* error = NULL;
    
    float* output_data = (float*)larodGetTensorBuffer(output, &error);
    size_t output_size = larodGetTensorSize(output, &error);
    
    int num_detections = 0;
    int num_boxes = output_size / (5 + 80); // 5 + num_classes
    
    for (int i = 0; i < num_boxes && num_detections < max_detections; i++) {
        float* box = &output_data[i * (5 + 80)];
        
        // Extract box coordinates and confidence
        float x = box[0];
        float y = box[1];
        float w = box[2];
        float h = box[3];
        float obj_conf = box[4];
        
        // Find best class
        int best_class = 0;
        float best_score = 0.0f;
        
        for (int c = 0; c < 80; c++) {
            float class_score = box[5 + c] * obj_conf;
            if (class_score > best_score) {
                best_score = class_score;
                best_class = c;
            }
        }
        
        // Filter by confidence threshold
        if (best_score > confidence_threshold) {
            detections[num_detections].x = x - w/2;
            detections[num_detections].y = y - h/2;
            detections[num_detections].width = w;
            detections[num_detections].height = h;
            detections[num_detections].class_id = best_class;
            detections[num_detections].confidence = best_score;
            
            // Set class name (simplified)
            snprintf(detections[num_detections].class_name, 64, 
                    "Class_%d", best_class);
            
            num_detections++;
        }
    }
    
    return num_detections;
}

// Draw detections using overlay API
void draw_detections(Detection* detections, int num_detections,
                    int overlay_width, int overlay_height) {
    GError* error = NULL;
    
    for (int i = 0; i < num_detections; i++) {
        Detection* det = &detections[i];
        
        // Convert normalized coordinates to pixel coordinates
        int x = (int)(det->x * overlay_width);
        int y = (int)(det->y * overlay_height);
        int w = (int)(det->width * overlay_width);
        int h = (int)(det->height * overlay_height);
        
        // Draw bounding box (using Cairo in render callback)
        // This would be called from the overlay render callback
        printf("Detection: %s (%.2f) at [%d,%d,%d,%d]\n",
               det->class_name, det->confidence, x, y, w, h);
    }
}

// Overlay render callback
static void render_overlay_cb(gpointer rendering_context,
                             gint id,
                             struct axoverlay_stream_data* stream,
                             enum axoverlay_position_type postype,
                             gfloat overlay_x,
                             gfloat overlay_y,
                             gint overlay_width,
                             gint overlay_height,
                             gpointer user_data) {
    
    Detection* detections = (Detection*)user_data;
    cairo_t* cr = (cairo_t*)rendering_context;
    
    // Set drawing style
    cairo_set_source_rgb(cr, 0.0, 1.0, 0.0); // Green
    cairo_set_line_width(cr, 2.0);
    
    // Draw each detection
    for (int i = 0; i < MAX_DETECTIONS; i++) {
        if (detections[i].confidence > 0.0f) {
            // Convert normalized to pixel coordinates
            int x = (int)(detections[i].x * overlay_width);
            int y = (int)(detections[i].y * overlay_height);
            int w = (int)(detections[i].width * overlay_width);
            int h = (int)(detections[i].height * overlay_height);
            
            // Draw bounding box
            cairo_rectangle(cr, x, y, w, h);
            cairo_stroke(cr);
            
            // Draw label
            cairo_move_to(cr, x, y - 5);
            cairo_show_text(cr, detections[i].class_name);
        }
    }
}
Lab 5
Parameter Hot-reload
Expert
Dynamic model switching and parameter updates using the Parameter API
Objective: Implement dynamic model switching and parameter updates without restarting the application

Hot-reload Features:

  • Dynamic model loading and unloading
  • Parameter API integration for configuration
  • Thread-safe model switching
  • Graceful error handling and rollback

Implementation:

#include <larod.h>
#include <axparameter.h>
#include <pthread.h>

typedef struct {
    larodConnection* conn;
    larodModel* current_model;
    char current_model_path[256];
    pthread_mutex_t model_mutex;
    AXParameter* param_handler;
} ModelManager;

// Parameter callback for model path changes
static void model_path_callback(const char* name, const char* value, void* user_data) {
    ModelManager* mgr = (ModelManager*)user_data;
    
    printf("Model path parameter changed: %s\n", value);
    
    // Attempt to load new model
    if (load_new_model(mgr, value) == 0) {
        printf("Successfully switched to new model: %s\n", value);
    } else {
        printf("Failed to load new model, keeping current model\n");
    }
}

// Load new model with thread safety
int load_new_model(ModelManager* mgr, const char* model_path) {
    larodError* error = NULL;
    larodModel* new_model = NULL;
    
    // Load new model
    new_model = larodLoadModel(mgr->conn, model_path,
                              LAROD_DEVICE_ID_CPU, LAROD_ACCESS_PRIVATE,
                              "", NULL, &error);
    
    if (!new_model) {
        fprintf(stderr, "Failed to load model %s: %s\n", 
                model_path, larodGetErrorString(error));
        return -1;
    }
    
    // Thread-safe model switching
    pthread_mutex_lock(&mgr->model_mutex);
    
    // Unload old model
    if (mgr->current_model) {
        larodUnloadModel(&mgr->current_model, NULL);
    }
    
    // Switch to new model
    mgr->current_model = new_model;
    strncpy(mgr->current_model_path, model_path, sizeof(mgr->current_model_path) - 1);
    
    pthread_mutex_unlock(&mgr->model_mutex);
    
    return 0;
}

// Thread-safe inference with current model
int run_inference_safe(ModelManager* mgr, larodTensor* input, larodTensor* output) {
    larodError* error = NULL;
    int result = -1;
    
    pthread_mutex_lock(&mgr->model_mutex);
    
    if (mgr->current_model) {
        larodTensor* inputs[] = {input};
        larodTensor* outputs[] = {output};
        
        if (larodRunInference(mgr->current_model, inputs, 1, outputs, 1, &error)) {
            result = 0;
        } else {
            fprintf(stderr, "Inference failed: %s\n", larodGetErrorString(error));
        }
    }
    
    pthread_mutex_unlock(&mgr->model_mutex);
    
    return result;
}

// Initialize model manager with parameter API
ModelManager* init_model_manager() {
    ModelManager* mgr = calloc(1, sizeof(ModelManager));
    AXParameterError param_error = AX_PARAMETER_OK;
    
    // Initialize Larod connection
    larodError* error = NULL;
    mgr->conn = larodConnect(&error);
    
    // Initialize mutex
    pthread_mutex_init(&mgr->model_mutex, NULL);
    
    // Initialize parameter handler
    mgr->param_handler = ax_parameter_new("MyMLApp", &param_error);
    
    // Register parameter callback for model path
    ax_parameter_register_callback(mgr->param_handler, "ModelPath",
                                  model_path_callback, mgr, &param_error);
    
    // Set default model path
    const char* default_model = "/usr/local/packages/MyMLApp/model.tflite";
    ax_parameter_set_string(mgr->param_handler, "ModelPath", 
                           default_model, &param_error);
    
    // Load initial model
    load_new_model(mgr, default_model);
    
    return mgr;
}

// Cleanup model manager
void cleanup_model_manager(ModelManager* mgr) {
    if (!mgr) return;
    
    // Cleanup parameter handler
    if (mgr->param_handler) {
        ax_parameter_free(mgr->param_handler);
    }
    
    // Cleanup model
    pthread_mutex_lock(&mgr->model_mutex);
    if (mgr->current_model) {
        larodUnloadModel(&mgr->current_model, NULL);
    }
    pthread_mutex_unlock(&mgr->model_mutex);
    
    // Cleanup Larod connection
    if (mgr->conn) {
        larodDisconnect(&mgr->conn, NULL);
    }
    
    // Cleanup mutex
    pthread_mutex_destroy(&mgr->model_mutex);
    
    free(mgr);
}
